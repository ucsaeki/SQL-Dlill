CREATE TABLE
	employee
	(
		s_id CHAR(7) NOT NULL,
		l_name VARCHAR(20),
		f_name VARCHAR(20),
		l_name_kana VARCHAR(100),
		f_name_kana VARCHAR(100),
		sex SMALLINT DEFAULT 1,
		class VARCHAR(20),
		depart_id CHAR(3),
		b_id CHAR(7) DEFAULT NULL,
		last_update DATE,
		retired SMALLINT DEFAULT 0
	)
;
