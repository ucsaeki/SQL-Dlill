CREATE TABLE
	usr
	(
		user_id CHAR(7),
		l_name VARCHAR(20),
		f_name VARCHAR(20),
		l_name_kana VARCHAR(100),
		f_name_kana VARCHAR(100),
		prefecture VARCHAR(15),
		city VARCHAR(20),
		o_address VARCHAR(100),
		tel VARCHAR(20),
		email VARCHAR(255),
		PRIMARY KEY (user_id)
	)
;
